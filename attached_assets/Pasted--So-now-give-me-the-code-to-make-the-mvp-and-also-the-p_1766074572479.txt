2. Next.js API routes for pilots
ts
// app/api/pilots/route.ts (Next.js App Router)
import { NextRequest, NextResponse } from "next/server";
import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import { prisma } from "@/lib/prisma";

export async function GET(req: NextRequest) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.email) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const user = await prisma.user.findUnique({
    where: { email: session.user.email },
    include: { pilots: true },
  });

  return NextResponse.json({ pilots: user?.pilots ?? [] });
}

export async function POST(req: NextRequest) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.email) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const body = await req.json();
  const { name, type, orgName, region, primaryObjective, majorityLogicDesc, qiLogicDesc } = body;

  const user = await prisma.user.upsert({
    where: { email: session.user.email },
    update: {},
    create: { email: session.user.email },
  });

  const pilot = await prisma.pilotProject.create({
    data: {
      name,
      type,
      orgName,
      region,
      primaryObjective,
      majorityLogicDesc,
      qiLogicDesc,
      ownerId: user.id,
    },
  });

  return NextResponse.json({ pilot }, { status: 201 });
}
3. Simple “Run Simulation” mock
You can start with a fake “A vs. B” result generator that mirrors the Magnum Opus story: Scenario A (Majority Rules) vs Scenario B (Qi Logic) with delta metrics (innovation up, liability down).​

ts
// app/api/pilots/[id]/run/route.ts
import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/lib/prisma";
import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";

export async function POST(
  req: NextRequest,
  { params }: { params: { id: string } }
) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.email) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const pilot = await prisma.pilotProject.findUnique({
    where: { id: params.id },
  });

  if (!pilot) {
    return NextResponse.json({ error: "Not found" }, { status: 404 });
  }

  // TODO: replace with real Omniverse integration
  const result = {
    scenarioA: {
      label: "Majority Logic",
      innovationIndex: 1.0,
      burnoutIndex: 0.8,
      liabilityIndex: 0.7,
    },
    scenarioB: {
      label: "Qi Logic",
      innovationIndex: 1.4, // +40%
      burnoutIndex: 0.3,    // lower is better
      liabilityIndex: 0.1,  // -90%
    },
  };

  // You could persist this in a separate SimulationResult table
  return NextResponse.json({ pilotId: pilot.id, result });
}
